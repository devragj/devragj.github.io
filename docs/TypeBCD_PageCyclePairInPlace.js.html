"use strict";

/**
 * The configuration object for the webpage CyclesAnimatePairInPlace.html.
 */

 let pageCyclePairInPlace = null;
 $(function() {
         pageCyclePairInPlace = new PageCycles({
                 clearList: ['comment', 'tableauPairRender'],
                 parse: ParameterDominoRS.parse,
                 generateParameter: ParameterDominoRS.generateParameter,
                 getDrawable: function(tableauPair, type, animationDuration, inPlace, unboxedOnly) {
                         return new TableauPairCycles({type, left: tableauPair.left,
                                 right: tableauPair.right, animationDuration, inPlace, unboxedOnly});
                 },
                 draw: function(tableauPair) {tableauPair.drawWithSpecialSquares();},
                 inPlace: true,
                 type: true
                 });
 });
